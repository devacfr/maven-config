Dependency Information

Apache Maven

<dependency>
  <groupId>org.cfr</groupId>
  <artifactId>maven-checker</artifactId>
  <version>6</version>
</dependency>

Apache Buildr

'org.cfr:maven-checker:jar:6'

Apache Ant

<dependency org="org.cfr" name="maven-checker" rev="6">
  <artifact name="maven-checker" type="jar" />
</dependency>

Groovy Grape

@Grapes(
@Grab(group='org.cfr', module='maven-checker', version='6')
)

Grails

compile 'org.cfr:maven-checker:6'

Leiningen

[org.cfr/maven-checker "6"]

SBT

libraryDependencies += "org.cfr" %% "maven-checker" % "6"
